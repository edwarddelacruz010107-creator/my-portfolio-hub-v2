# ─── Google OAuth production setup ─────────────────────────────
# Create/select a Web application OAuth client in Google Cloud Console.
# Do NOT commit real secrets. Set these values only in your production host.

GOOGLE_OAUTH_REDIRECT_URI=https://myportfoliohub.online/auth/google/callback
